\section{Misc Problem 6.27}
\\If \( P(A) = 6/11 , P(B) = 5/11\) and \(P(A + B) = 7/11\) find
\\i. \(P(AB)\)
\\ii. \(P(A|B)\)
\\iii. \(P(B|A)\)
\section{Solution}
i.
\begin{align}
P(A + B) = P(A) + P(B) - P(AB)
\end{align}
Substituting the values,
\begin{align}
\frac{7}{11} = \frac{6}{11} + \frac{5}{11} - P(AB)\\
 P(AB) = 1 - \frac{7}{11} = \frac{4}{11}
\end{align}
ii.
\begin{align}
P(A|B) = \frac{P(AB)}{P(B)} = \frac{4/11}{5/11} = \frac{4}{5}
\end{align}
iii.
\begin{align}
P(B|A) = \frac{P(AB)}{P(A)} = \frac{4/11}{6/11} = \frac{2}{3}
\end{align}
